class Main {
    
    static int n1=0,n2=1,sum=0;
    public static void fibo (int value){
        
        System.out.print(n1 + " " + n2);
        for(int i=2;i<value;i++){
            sum = n1+n2;
            n1=n2;
            n2=sum;
            System.out.print(" " + sum);
            
        }
    }
    public static void main(String[] args) {
        System.out.println("Try programiz.pro");
        Main.fibo(20);
        
    }
}
